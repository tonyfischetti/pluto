FOR-EACH

(FOR-EACH/LIST '(A B C)
  (FORMAT T "~A -> ~A~%" INDEX! VALUE!))

Outputs:

1 -> A
2 -> B
3 -> C

FOR-EACH

(FOR-EACH/LIST '(A B C D E)
  (IF (> INDEX! 2)
      (BREAK!))
  (FORMAT T "~A~%" VALUE!))

Outputs:

A
B

FOR-EACH

(FOR-EACH/LIST '(A B C D E)
  (IF (= INDEX! 3)
      (CONTINUE!))
  (FORMAT T "~A~%" VALUE!))

Outputs:

A
B
D
E

1 -> we gotta celebrate diversity

FOR-EACH

(LET ((TMP (MAKE-HASH-TABLE)))
  (SETF (GETHASH 'GREEN TMP) "veridian")
  (SETF (GETHASH 'RED TMP) "cadmium")
  (FOR-EACH TMP
    (FORMAT T "~A -> ~A~%" KEY! VALUE!)))

Outputs:

GREEN -> veridian
RED -> cadmium
